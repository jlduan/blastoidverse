create_gt <- function(x) {
    x |>
        gt::gt() |>
        gt::data_color(
            columns = c(median_umis),
            fn = scales::col_numeric(
                palette = c(
                    "green", "orange", "red"
                ),
                domain = NULL
            )
        ) |>
        gt::fmt_number(
            columns = c(num_cells),
            sep_mark = ",",
            decimals = 0,
            suffixing = FALSE
        ) |>
        gt::fmt_number(
            columns = c(median_umis),
            sep_mark = ",",
            decimals = 1,
            suffixing = FALSE
        ) |>
        gt::grand_summary_rows(
            columns = c(num_cells),
            fns = c(
                Sum = ~ sum(.)
            ),
            fmt = ~ gt::fmt_number(., decimals = 0, use_seps = TRUE),
            suffixing = FALSE
        )
}
